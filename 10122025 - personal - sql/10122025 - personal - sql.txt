SELECT
    employee_name
FROM
    employee
WHERE
    employee_salary > (
                SELECT 
                    avg(employee_salary)
                FROM
                    employee
    )




SELECT
    employee_name,
    (
        SELECT
            department_name
        FROM
            department AS d
        WHERE
            d.id = e.dept_id
    ) AS employee_department_name
FROM
    employee AS e;






SELECT
    employee_name
FROM
    employee AS e
JOIN
    (
        SELECT
            dept_id,
            avg(employee_salary) AS average_salary
        FROM
            employee
        GROUP BY
            dept_id
    ) as dt
    ON
        dt.dept_id = e.dept_id
WHERE
    employee_salary > dt.average_salary
